\section{Introduction}

\begin{frame}{Gestion de versions, qu'est-ce que c'est ?}
	\begin{itemize}
		\item Gestion des changements sur un projet
		\note[item]{La gestion de version consiste à gérer les changements sur un fichier ou un ensemble de fichiers (un projet).}
		\item Permet de :
			\begin{itemize}
				\item Revoir les changements déjà fait
				
				\item Annuler des changements problématiques
				\note[item]{Cela permet de revoir les changements, voir de les annuler.}
				\item Collaborer avec plusieurs personnes sur le même projet
				\note[item]{On peut facilement gérer des modifications qui viennent de différentes personnes en même temps}
			\end{itemize}
		\item Il existe de nombreux systèmes
			\begin{itemize}
				\item Copier-coller le dossier et le renommer "projet-vx"
				\item SVN
				\item Git
			\end{itemize}
	\end{itemize}
\end{frame}
	
\begin{frame}{Git}
	\begin{itemize}
		\item Créé en 2005 par Linus Torvalds pour le développement du Kernel Linux
		\item Appartient à la catégorie des systèmes de contrôle de version \textbf{distribué}
		\note[item]{C'est un système distribué. Cela veut dire que chaque utilisateur possède une copie complète du projet. Ce qui permet de ne pas dépendre d'un serveur central.}
		\item C'est un des systèmes les plus utilisés dans le monde, encore plus l'open source
	\end{itemize}
\end{frame}

\begin{frame}{Git is not GitHub}
\begin{itemize}
	\item Git est l'outil de base
	\item Un serveur peut être connecté pour collaborer
	\item GitHub fourni un service de serveur
\end{itemize}

\end{frame}

	  \item<3->

	  \item<4->
		Second item.
	  \end{itemize}

	  en utilisant en général la commande \texttt{uncover} :
	  \begin{itemize}
		\uncover<5->{\item
		  Premier item.}
		\uncover<6->{\item
		  Second item.}
	  \end{itemize}
	\end{itemize}
